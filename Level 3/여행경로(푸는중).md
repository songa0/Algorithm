function solution(tickets) {
    var answer = [];
    let airport = new Map();
    let used = new Array(tickets.length).fill(false);
    
    tickets.forEach((item, idx)=>{
        if(airport.has(item[0])){
            const arr = airport.get(item[0]);
            arr.push([item[1], idx]);
            airport.set(item[0],arr);
        }else{
            airport.set(item[0], [[item[1], idx]]);
        }
    });
    
    
    
    return answer;
}
