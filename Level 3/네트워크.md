  function solution(n, computers) {
    var answer = 0;
    let visited = new Array(computers.length);
    let group = new Array(n);
    group.fill(0);
    for(let i =0; i<n; i++){
        visited[i] = new Array(computers[0].length);
        visited[i].fill(0);
    }
    let networkNum = 0;
    
    const visitNet = (i,j) =>{
        console.log('방문',i,j, networkNum);
        visited[i][j] = networkNum;
        visited[j][i] = networkNum;
        group[i] = networkNum;
        group[j] = networkNum;
        
        for(let k = 0; k<computers[0].length; k++){
            if(j===k) continue;
            if(computers[j][k] ===1 && !visited[j][k]){
                visitNet(j,k);
            }
        }
        
    }
    
    for(let i =0; i<computers.length; i++){
        for(let j = i+1; j<computers[0].length; j++){
            if(computers[i][j]==1){
                if(!visited[i][j] && !visited[j][i]){
                    networkNum++;
                    visitNet(i,j);
                   
                }
                   
             }
        }
    }
   
    group.forEach(item =>{
        if(!item) networkNum++;
    })
    return networkNum;
}
