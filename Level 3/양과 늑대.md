## 내 풀이  
```javascript
function solution(info, edges) {
    var answer = 0;
    let graph = [];
    
    //graph 생성
    for(const [parent, child] of edges){
        if(graph[parent]){
            graph[parent].push(child);
        }else{
            graph[parent] = [child];
        }
    }
   // console.log(graph);
    
    dfs(0,0,1, [true]);
    
    function dfs(current, wolf, sheep, visited){
        //console.log(current, "방문", wolf, sheep);
        answer = Math.max(sheep, answer);
        if(wolf == sheep) {
            return;
        }
        
        if(graph[current]){
            for(const child of graph[current]){
                //console.log("부모 자식", current, child);
                if(!visited[child] && info[child] && wolf+1 <sheep){//늑대
                    visited[child] = true;
                    dfs(child, ++wolf, sheep, visited);
                    
                   
                }
                if(!visited[child] && !info[child]){
                    visited[child] = true;
                    dfs(child, wolf, ++sheep, visited);
                    
                }
            }
        }
        
        
    }
    
    return answer;
}
```
