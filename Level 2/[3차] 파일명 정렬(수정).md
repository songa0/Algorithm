## 내 풀이  
```javascript
function solution(files) {
    
    
    files = files.sort((a,b)=> {
        let startOfNumA = a.search(/[0-9]/g);
        let startOfNumB = b.search(/[0-9]/g);
        let headerA = a.substring(0, startOfNumA).toLowerCase();
        let headerB = b.substring(0, startOfNumB).toLowerCase();
        
        if(headerA > headerB) return 1;
        else if(headerA == headerB) {
            let exceptHdrA = a.substring(startOfNumA);
            let exceptHdrB = b.substring(startOfNumB);
            let numberA = parseInt(exceptHdrA.substring(0, exceptHdrA.search(/[^0-9]/g)));
            let numberB = parseInt(exceptHdrB.substring(0, exceptHdrB.search(/[^0-9]/g)));
            
            return numberA-numberB;
        }
        else return -1;       
    });
    
    return files;
}


```
