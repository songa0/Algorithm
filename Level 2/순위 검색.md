## 내풀이
   조합을 만든 후, 바이너리 서치로 주어진 점수 이상의 값을 가진 것들을 찾아야한다.  
   일단 정확성만 다 맞게 코딩하였다.  
```javascript
function solution(info, query) {
    var answer = [];
    let map = new Map();
    
    function setMap(depth,str, arr){
        if(depth>3){
            if(map.has(str)){
                let value = map.get(str);
                value.push(arr[4]);
                map.set(str, value);
            }else{
                map.set(str, [arr[4]]);
            }
            return;
        }
        
        setMap(depth+1, `${str} ${arr[depth]}`, arr);
        setMap(depth+1, `${str} -`, arr);
        
    }
    
    for(let i =0; i<info.length; i++){
       let arr = info[i].split(' ');     
       setMap(1, arr[0], arr);
       setMap(1, '-',arr);
    }
    
    for(let key of map.keys()){
        let value = map.get(key);
        value.sort(function(a,b){return a-b;});
        
        map.set(key, value);
       // console.log(key ,value)
    }
    for(let i = 0; i<query.length; i++){
        let q = query[i].replace(/and /gi, '').split(' ');
        let score = parseInt(q.pop());
        let key = q.join(" ");
        let arr = map.get(key);
        if(map.has(key)){
            let count = 0;
            for(let j =0; j<arr.length; j++){
                if(parseInt(arr[j])>=score){
                    count++;
                    
                }
            }
            //console.log(key, arr, arr.length, idx);
            answer.push(count);
            
        }else{
            answer.push(0);
        }
        
        
    }
    
    return answer;
}



```
