## 내 풀이 (실패)

```javascript
function solution(people, limit) {
    var answer = 0;
    people = people.sort((a,b) => parseInt(a)-parseInt(b));
    //console.log(people);
    let sum = 0;
    for(let i =0; i<people.length-1; i++){
        if(sum + people[i] <=limit){
            sum +=people[i];
        }else{
            answer ++;
            i--;
            sum = 0;
        }         
    }
    
    if(sum + people[people.length-1] <=limit){
           answer +=1;
        }else{
            answer +=2
        }         
    return answer;
}

```
