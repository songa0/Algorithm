## 내 풀이 (효율성 실패로 다시 풀어야함)
```javascript
const num = ['1','2','4'];
let  answer = '';
function solution(n) {
    
    if(n<=3) return num[n-1];
    else {dp(num, 3,n);
         return answer;}
       
}
const dp = (original, current, n) =>{
       
        const arr = new Array();
        let cnt = current;
        for(let i =0; i<3; i++){
            for(let j = 0; j<original.length; j++){
                arr.push(num[i]+original[j]);
                cnt +=1;   
                
                if(cnt=== n){answer =  arr[arr.length-1]; return;}
                
            }
        }
        
        dp(arr, cnt, n);
    
    }


```
